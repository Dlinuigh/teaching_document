\section{力学}
\subsection{运动学}
\begin{formulatab}{运动学}
  \multicolumn{2}{l}{\textbf{匀速直线运动}}\\
  速度&$v=\frac{x}{t}$\\
  \multicolumn{2}{l}{\textbf{匀变速直线运动}}\\
  加速度&$a=\bar{a}$\\
  速度&$v=v_0+at$\\
  位移&$x=v_0 t + \frac12 a t^2$\\
  位移&$x=\frac12(v+v_0)t$\\
  速度-位移关系&$v^2-v_0^2=2ax$\\
\end{formulatab}

位移和路程、速度和平均速度、速率和平均速率、加速度是运动学中的基础物理量。

\textbf{曲线运动判定}

曲线运动发生在合力与速度不共线时。当力与速度夹角小于 $90^{\circ}$ 时加速，等于 $90^{\circ}$ 时匀速，超过 $90^{\circ}$ 时减速。

\subsection{平抛运动}
\begin{formulatab}{平抛运动}
  水平位移&$x=v_0 t$\\
  竖直位移&$y=\frac12 a t^2$\\
  竖直速度&$v_y=a t=\sqrt{\frac{2y}{a}}=v_0\tan\theta$\\
  合速度&$v=\sqrt{v_y^2+v_0^2}=\frac{v_0}{\cos\theta}$\\
\end{formulatab}

平抛运动是指初速度与恒定合力垂直的运动。

\subsection{匀速圆周运动}
\begin{formulatab}{匀速圆周运动}
  向心力&$F_{\text{向}}=m\frac{v^2}{r}$\\
  向心力&$F_{\text{向}}=ma_{\text{向}}$\\
  线速度&$v=\omega r$\\
  角速度&$\omega=\frac{2\pi}{T}$\\
\end{formulatab}

匀速圆周运动要求向心力恒定，速度大小不变，方向不断变化。

\subsection{万有引力}
\begin{formulatab}{万有引力}
  \multicolumn{2}{l}{\textbf{万有引力}}\\
  万有引力&$F=\frac{GMm}{R^2}$\\
  \multicolumn{2}{l}{\textbf{开普勒三大定律}}\\
  周期定律&$\frac{a^3}{T^2}=k$\\
  \multicolumn{2}{l}{\textbf{黄金代换}}\\
  黄金代换&$\frac{GMm}{R^2}=mg$\\
  \multicolumn{2}{l}{\textbf{宇宙速度}}\\
  第一宇宙速度&$v=\sqrt{\frac{GM}{R}}$\\
\end{formulatab}

\textbf{开普勒定律补充}

\begin{itemize}
  \item 轨道定律：行星绕太阳的轨道是椭圆。
  \item 面积定律：行星与太阳的连线在单位时间内扫过的面积相等，近太阳时快、远太阳时慢。
\end{itemize}

从低轨道到高轨道需要加速。

\subsection{静力学}
\begin{formulatab}{静力学}
  \multicolumn{2}{l}{\textbf{常见的力}}\\
  重力&$G=mg$\\
  摩擦力&$f=\mu N$\\
  弹簧弹力&$F=kx$\\
\end{formulatab}

\textbf{常见的力}

\begin{itemize}
  \item 重力：$G=mg$
  \item 弹力：弹簧、支撑面等产生的反作用力。
  \item 摩擦力：滑动摩擦力大小为 $f=\mu N$，方向与相对运动方向相反；静摩擦力大小与实际情况有关，方向阻碍相对运动趋势。
  \item 弹簧弹力：$F=kx$，弹簧压缩时弹力向外，伸长时弹力向内。
\end{itemize}

\subsection{动力学}
\begin{formulatab}{动力学}
  \multicolumn{2}{l}{\textbf{牛顿运动定律}}\\
  牛顿第二定律&$F=ma$\\
  \multicolumn{2}{l}{\textbf{空气阻力}}\\
  空气阻力&$f=kv$\\
\end{formulatab}

\textbf{牛顿运动定律}

动力学核心是牛顿第二定律$F=ma$，合力决定加速度。

\textbf{空气阻力}

空气阻力方向与速度方向相反。
